bitwise-Op Beispiel 

/* Studienbrief04 - bit_operation*/
#include <stdio.h>
int main(int argc, char* argv[])
{
 unsigned char a = 45; /* 00101101 - 1 Byte */
 unsigned char b = 170; /* 10101010 - 1 Byte */
 unsigned char result = 0;
 /* bitwise AND */
 result = a & b;
 printf("%d\n", (int)result);
 /* 40 - binary 00101000 */
 /* bitwise OR */
 result = a | b;
 printf("%d\n", (int)result);
 /* 175 - binary 10101111 */
 /* bitwise XOR */
 result = a ^ b;
 printf("%d\n", (int)result);
 /* 135 - binary 10000111 */
 /* complement */
 result = ~a;
 printf("%d\n", (int)result);
 /* 210 - binary 11010010 */
 /* bitwise shift right */
 result = a;
 result >>= 3;
 printf("%d\n", (int)result);
 /* 5 - binary 00000101 */
 /* bitwise shift left */
 result = a;
 result <<= 3;
 printf("%d\n", (int)result);
 /* 104 - binary 01101000 */
 return 0;
}
