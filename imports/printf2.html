printf floating point numbers

Code Result 
printf("'%.1f'", 10.3456);
 
'10.3'
 
printf("'%.2f'", 10.3456);
 
'10.35'
 
printf("'%8.2f'", 10.3456);
 
'   10.35'
 
printf("'%8.4f'", 10.3456);
 
' 10.3456'
 
printf("'%08.2f'", 10.3456);
 
'00010.35'
 
printf("'%-8.2f'", 10.3456);
 
'10.35   '
 
printf("'%-8.2f'", 101234567.3456);
 
'101234567.35'
