1)Assoziativität
"=>" bedeutet: Ausführung erst des linken Operators, dann des rechten Operators a OP1 b OP2 c -> OP1 dann OP2 (1 + 2 - 3 <=> (1 + 2) - 3)
"<=" bedeutet: Ausführung erst des rechten Operators, dann des linken Operators a OP1 b OP2 c -> OP2 dann OP1 (!-a <=> !(-a))
x = 100
x++        //x = 100 + 1; x = 101
x += 2     //x = 100 + 2; x = 102
x -= 2     //x = 100 - 2; x = 98
x *= x - 1 //x = 100 * (100 - 1); x = 9900
x *= x--   //x = (100 * 100) - 1; x = 9999
x += x++   //x = (100 + 100) + 1; x = 201
x *= --x   //x = (100 - 1) * (100 - 1); x = 9801
x += ++x   //x = (100 + 1) + (100 + 1); x = 202
